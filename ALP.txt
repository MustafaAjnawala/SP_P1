MACRO SWAPPING &i1,&i2
MOV AL,grades[&i1]
CMP AL,grades[&i2]
JAE SKIP
XCHG grades[&i1],grades[&i2]
XCHG students[&i1],students[&i2]
SKIP:
MEND

START
MOV CX,0
OUTER_LOOP: MOV SI,CX
INNER_LOOP:
MOV DI, SI
INC DI
SWAPPING SI,DI
INC SI
CMP SI,3
JLE INNER_LOOP
INC CX
CMP CX,3
JLE OUTER_LOOP
students: DB 1,2,3,4;
grades: DS 5,1,6,2;
END
